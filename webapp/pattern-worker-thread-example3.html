Worker Thread - example


    
    public class Main {
        public static void main( String[] args ) {
            RequestQueue queue = new RequestQueue();
            
            for ( int i = 0 ; i < 10 ; ++i ) {
                new Worker( queue ).start();
            }
            
            for ( int i = 0 ; i < 1000 ; ++i ) {
                queue.putRequest( new Request( "Message" + i ) );
            }
            
        }
    }
