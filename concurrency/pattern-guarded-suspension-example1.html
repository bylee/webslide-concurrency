Guarded suspension - example


    public class Client extends Thread {
        protected final Random r = new Random( System.currentTimeMillis() );
        
        protected final RequestQueue queue;
        
        public Client( String name, RequestQueue queue ) {
            super( name );
            this.queue = queue;
        }
        
        public void run() {
            for ( int i = 0 ; i<100 ; ++i ) {
                Request request = new Request( "No." + i, getName() );
                queue.put( request );
                trySleep( r.nextInt( 1000 ) );
            }
        }
    }
