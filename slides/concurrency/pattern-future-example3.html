Future - example



public class Future implements Data {

    protected Data data;
    
    synchronized public void setData( Data data ) {
        this.data = data;
        notifyAll();
    }
    
    synchronized public String getContent() {
    
        while( null == data ) {
            try {
                wait();
            } catch ( InterruptedExceptioni e ) {
            }
        }
        
        return data.getResult();
    }
}
